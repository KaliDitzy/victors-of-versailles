on_actions = {
	on_startup = {
		effect = {
			set_global_flag = SGF_pre_election
			set_global_flag = {
				flag = days_since_last_great_war
				value = 1984
			}

			every_country = {
				limit = {
					NOT = {
						original_tag = SGF
					}
				}

				if = {
					limit = {
						check_variable = { next_suffrage_will_base = 0 }
					}
					set_variable = { next_suffrage_will_base = 0.2 }
				}
				if = {
					limit = {
						check_variable = { previous_suffrage_will_base = 0 }
					}
					set_variable = { previous_suffrage_will_base = 0.4 }
				}
				calculate_reform_will = yes

				load_generic_units = yes
			}
		}
	}
}

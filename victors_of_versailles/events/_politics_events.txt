add_namespace = next_suffrage_law

country_event = {
	id = next_suffrage_law.1

	trigger = {
		has_country_flag = next_suffrage_in_progress
		NOT = { check_variable = { passing_law_time < 7 } }
	}

	mean_time_to_happen = {
		days = 14
	}

	hidden = yes

	immediate = {
		if = {
			limit = {
				NOT = { check_variable = { law_progress > 0 } }
			}
			clr_country_flag = passing_law
			clr_country_flag = next_suffrage_in_progress
			country_event = next_suffrage_law.3
		}
		else_if = {
			limit = {
				NOT = { check_variable = { law_progress < 1 } }
			}
			inc_suffrage_law = yes
			clr_country_flag = passing_law
			clr_country_flag = next_suffrage_in_progress
			country_event = next_suffrage_law.2
			reduce_consciousness_from_reform = yes
		}
		else = {
			random_list = {
				20 = {
					country_event = next_suffrage_law.4
				}
				40 = {
					country_event = next_suffrage_law.5
				}
			}
		}
	}
}
